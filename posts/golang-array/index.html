Golang 基础：array

2019年7月10日
| Golang

array

  • 必须指定长度
  • 传递数组指针修改数组内元素才有效
package main

import "fmt"

func main() {
	//var arr [5]int
	arr := [5]int{
		0: 1,
	}

	// 同类型，同长度才可相互赋值
	// var arr1 [4]int
	// arr1 = arr

	fmt.Printf("%T(%#v)\n", arr, arr)
	fmt.Printf("%T(%#v)\n", arr[0], arr[0])
	fmt.Printf("%T(%#v)\n", &arr[0], &arr[0])

	for i, v := range arr {
		fmt.Printf("索引:%d,值:%d\n", i, v)
	}

	// 指针数组
	ptrArr := [5]*int{
		// 分配内存
		1: new(int),
		// 其他索引为 nil
	}
	fmt.Printf("%T(%#v)\n", ptrArr, ptrArr)
	fmt.Printf("%T(%#v)\n", ptrArr[1], ptrArr[1])
	fmt.Printf("%T(%#v)\n", *ptrArr[1], *ptrArr[1])

	// 分配了内存才能赋值
	*ptrArr[1] = 1

	fmt.Printf("%T(%#v)\n", ptrArr, ptrArr)
	fmt.Printf("%T(%#v)\n", ptrArr[1], ptrArr[1])
	fmt.Printf("%T(%#v)\n", *ptrArr[1], *ptrArr[1])

	// 传递数组，不会改变值
	// modify(arr)
	modify(&arr)
	fmt.Println("main", arr)
}

func modify(a *[5]int) {
	a[1] = 3
	fmt.Println("modify", *a)
}
